# Пример правил исключения для YD2DBX (при необходимости скопируйте и отредактируйте).
# Строки с # — комментарии.
#
# prefix:<path>    — не синхронизировать всё под этим префиксом пути на Диске
# contains:<seg>   — исключить пути, содержащие сегмент (например /tmp/)
# suffix:<строка>  — исключить пути, оканчивающиеся на эту строку

# Временные каталоги на любом уровне
contains:/tmp/
contains:/temp/

# Свои каталоги — добавьте под свою структуру Yandex Disk, например:
# prefix:/Архив/большие_файлы
# suffix:.dmg
